# Install virtualenv 

# clone Repository
    cd word_puzzle
    
# Create virtual environment using python3 

    virtualenv -p python3 .venv
    
    source .venv/bin/activate
    
# Install requirements

    pip install -r requirements.txt
    
# Run puzzle 
    sudo chmod +x puzzle.py
    python puzzle.py
    
# Run unittest

    sudo chmod +x test.py
    python test.py
